import Browser
import Html exposing (Html, button, div, text, button, pre)
import Html.Attributes exposing (class)
import Html.Events exposing (onClick)


type alias Model =
    { name : String
    , kids : Int
    , counter : Int
    }


initialModel : Model
initialModel =
    { name = "Amitai"
    , kids = 3
    , counter = 0
    }


type Msg
    = IncrementCounter
    | DecrementCounter


update : Msg -> Model -> Model
update msg model =
    case msg of
        IncrementCounter ->
            { model | counter = model.counter + 1 }

        DecrementCounter ->
            { model | counter = model.counter - 1 }


view : Model -> Html Msg
view model =
      div [ class "container" ]
        [ div [ class "row" ]
            [ div [] [ text <| "Name: " ++ model.name ]
            , div [] [ text <| "Kids num: " ++ String.fromInt model.kids ]
            , button [ onClick DecrementCounter ] [ text "-" ]
            , button [ onClick IncrementCounter ] [ text "+" ]
            , pre [] [ text <| String.fromInt model.counter ]
            ]
        ]




main : Program () Model Msg
main =
    Browser.sandbox
        { init = initialModel
        , view = view
        , update = update
        }
